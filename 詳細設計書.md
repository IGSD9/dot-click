### 1-3. Prisma Schema

```prisma
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  name      String
  role      String   @default("user")
  scores    Score[]
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Score {
  id         String   @id @default(cuid())
  userId     String
  user       User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  maxStage   Int      @default(1)
  totalScore Int      @default(0)
  updatedAt  DateTime @updatedAt

  @@unique([userId])
  @@index([totalScore(sort: Desc)])
  @@index([maxStage(sort: Desc)])
}
```

### 2-7. 衝突判定

矢が的の外周に到達した瞬間、新しい角度と既存矢の角度差を比較する。

```typescript
// src/lib/game/collision.ts
export function checkCollision(
  newAngle: number,
  existingArrows: Arrow[],
  threshold = 0.15
): boolean {
  for (const arrow of existingArrows) {
    const diff = angleDifference(newAngle, arrow.angle);
    if (diff < threshold) return true;
  }
  return false;
}

function angleDifference(a: number, b: number): number {
  const twoPi = Math.PI * 2;
  const na = ((a % twoPi) + twoPi) % twoPi;
  const nb = ((b % twoPi) + twoPi) % twoPi;
  const diff = Math.abs(na - nb);
  return Math.min(diff, twoPi - diff);
}
```

### 2-9. 入力（PC・スマホ両対応）

**方針:** `pointerdown` を主入力とし、PC ではスペースキーも受け付ける。  
1 イベント = 1 射出。ボタンやリンクへの誤射出を防ぐ。

| デバイス | 入力 | イベント | 条件 |
|---------|------|---------|------|
| PC | クリック | `pointerdown` | `phase === 'idle'` |
| PC | スペース | `keydown` (Space) | 同上。`e.preventDefault()` |
| スマホ | タップ | `pointerdown` | 同上。画面どこでも可 |

### 3-5. スコア計算

```
totalScore = maxStage × 1000 + min(streak, 50) × 10

例: ステージ 15、連続 8 クリア
  → 15 × 1000 + 8 × 10 = 15080
```

更新ルール:
- `maxStage`: 今回 > 保存値 のときだけ更新
- `totalScore`: 算出値 > 保存値 のときだけ更新

### 3-6. saveScore

**入力（Zod）**

```typescript
const SaveScoreSchema = z.object({
  maxStage: z.number().int().min(1).max(9999),
  streak: z.number().int().min(0).max(9999),
});
```

**処理**

```
1. セッション確認 → 未ログインなら { success: false }
2. totalScore 算出
3. 既存 Score 取得
4. なければ create / あれば max 比較して update
5. { success: true, score } を返す
```

**実装**

```typescript
'use server';

export async function saveScore(input: z.infer<typeof SaveScoreSchema>) {
  const parsed = SaveScoreSchema.parse(input);
  const session = await getServerSession();
  if (!session) return { success: false, reason: 'UNAUTHORIZED' };

  const totalScore =
    parsed.maxStage * 1000 + Math.min(parsed.streak, 50) * 10;

  const existing = await prisma.score.findUnique({
    where: { userId: session.user.id },
  });

  if (!existing) {
    const score = await prisma.score.create({
      data: {
        userId: session.user.id,
        maxStage: parsed.maxStage,
        totalScore,
      },
    });
    return { success: true, score };
  }

  const newMaxStage = Math.max(existing.maxStage, parsed.maxStage);
  const newTotalScore = Math.max(existing.totalScore, totalScore);

  if (newMaxStage === existing.maxStage && newTotalScore === existing.totalScore) {
    return { success: true, score: existing, updated: false };
  }

  const score = await prisma.score.update({
    where: { userId: session.user.id },
    data: { maxStage: newMaxStage, totalScore: newTotalScore },
  });
  return { success: true, score, updated: true };
}
```
